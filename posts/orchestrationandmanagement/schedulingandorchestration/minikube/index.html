Minikube

minikube

pre

brew install docker brew install docker-compose

brew install kubectl brew install hyperkit

install

brew install minikube

curl -LO https://storage.googleapis.com/minikube/releases/latest/minikube-darwin-amd64 sudo install minikube-darwin-amd64 /usr/local/bin/minikube

minikube config set cpus 1 minikube config set memory 4g

minikube config set memory 8192

minikube config set driver hyperkit

usage

minikube start –vm-driver=“kvm2” –container-runtime=containerd –docker-opt containerd=/var/run/containerd/containerd.sock –docker-env=“http_proxy=x.x.x.x:8080” –docker-env=“https_proxy=x.x.x.x:8080” start

minikube start –kubernetes-version=1.18.8 –driver=hyperkit –container-runtime=containerd –image-mirror-country=‘cn’

minikube start –kubernetes-version=1.18.8 –driver=hyperkit –container-runtime=containerd –registry-mirror=https://registry.docker-cn.com

minikube start –driver=hyperkit –container-runtime=containerd –image-repository=‘registry.cn-hangzhou.aliyuncs.com/google_containers’ –kubernetes-version=1.18.8 --container-runtime: docker containerd cri-o

minikube start –cpus 3 –memory 8192 –disk-size 20g –vm-driver=hyperkit

minikube start –cpus 2 –memory 4g –disk-size 20g –driver=hyperkit –mount –mount-string="/Users/will/go/src/github.com/Mikaelemmmm/go-zero-looklook:/test" –container-runtime=docker –image-mirror-country=‘cn’ –kubernetes-version=1.23.3

minikube start –force –cpus 2 –memory 4g –disk-size 20g –driver=hyperkit –mount –mount-string="/Users/will/go/src/github.com/Mikaelemmmm/go-zero-looklook:/test" –container-runtime=docker –image-mirror-country=‘cn’ –kubernetes-version=1.23.3

minikube mount .:/test

minikube start –vm-driver=hyperv –mount –mount-string=“C:\Users\AUser\AFolder:/minikube-host”

–registry-mirror=https://registry.docker-cn.com

–profile [new-profile-name]

–registry-mirror=https://index.docker.io/v1/ –registry-mirror=https://dockerhub.azk8s.cn

–image-repository=‘registry.cn-hangzhou.aliyuncs.com/google_containers’ –image-mirror-country=‘cn’

使用 –logtostderr –v=3 参数以打印相关日志，便于发现启动过程中的问题。 使用 –image-mirror-country=cn 参数以顺利的下载 k8s 相关镜像。 使用 –registry-mirror https://dockerhub.azk8s.cn 参数以指定 docker image 镜像 - 便于国内下载。

如果你在本地运行dnsmasq，在集群中可能会出现DNS解析失败。你可以卸载它或者在dnsmasq.conf中添加listen-address=192.168.64.1。

alias kubectl=“minikube kubectl –”

minikube kubectl get nodes minikube ssh

kubectl create deployment hello-minikube –image=k8s.gcr.io/echoserver:1.4 kubectl expose deployment hello-minikube –type=NodePort –port=8080

kubectl get services hello-minikube kubectl port-forward service/hello-minikube 7080:8080

minikube service hello-minikube

kubectl create deployment balanced –image=k8s.gcr.io/echoserver:1.4 kubectl expose deployment balanced –type=LoadBalancer –port=8080

minikube tunnel kubectl get services balanced

eval $(minikube docker-env) docker info

minikube addons enable ingress minikube ip curl http://ip

minikube addons enable ingress-dns

minikube addons enable metallb minikube addons configure metallb

brew install docker-credential-helper

minikube pause minikube unpause minikube stop

minikube addons list

minikube start -p aged –kubernetes-version=v1.16.1

docker run –rm -it -v /test:/inside busybox /bin/sh
